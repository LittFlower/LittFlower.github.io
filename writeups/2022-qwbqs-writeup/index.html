    opt[64] = 'u'; opt[-64] = 'd';

    vis[st] = true;
    q.push(st);
    ans[st] = 0;

    while (!q.empty()) {
        int now = q.front();
        q.pop();
        if (now == 511) { return ans[now]; }
        for (int i = 0; i < 6; ++i) {
            int nxt = now + dir[i];
            if (nxt < 0 || nxt > 511 || vis[nxt] || mp[nxt]) continue;
            vis[nxt] = true;
            ans[nxt] = ans[now] + 1;
            cout << opt[dir[i]];
            q.push(nxt);
        }
    }
}
